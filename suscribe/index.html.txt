  <script>
    const form = document.getElementById('subscribeForm');
    const backendUrl = "https://script.google.com/macros/s/AKfycbyl4GgcxBdYwfzjXhdZjcRxKhdHRrTm16S8zZLxfHOmRNqnrKYEbgnB0ECUt3uP8hPH7Q/exec";

    form.addEventListener('submit', function(e) {
      e.preventDefault();
      const formData = new FormData(form);

      fetch(backendUrl, {
        method: 'POST',
        mode: 'no-cors',
        body: new URLSearchParams({
          action: 'subscribe',
          email: formData.get('email'),
          name: formData.get('name')
        })
      }).then(() => {
        alert("¡Gracias por suscribirte!");
        form.reset();
      });
    });
