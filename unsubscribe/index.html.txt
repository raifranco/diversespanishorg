  <form id="unsubscribeForm">
    <input type="email" id="email" name="email" required placeholder="Tu correo...">
    <button type="submit">Desuscribirme</button>
  </form>

  <script>
    const form = document.getElementById('unsubscribeForm');
    const backendUrl = "https://script.google.com/macros/s/AKfycbyl4GgcxBdYwfzjXhdZjcRxKhdHRrTm16S8zZLxfHOmRNqnrKYEbgnB0ECUt3uP8hPH7Q/exec";

    form.addEventListener('submit', function(e) {
      e.preventDefault();
      const formData = new FormData(form);

      fetch(backendUrl, {
        method: 'POST',
        mode: 'no-cors',
        body: new URLSearchParams({
          action: 'unsubscribe',
          email: formData.get('email')
        })
      }).then(() => {
        window.location.href = "success-unsubscribe.html";
      });
    });
  </script>
